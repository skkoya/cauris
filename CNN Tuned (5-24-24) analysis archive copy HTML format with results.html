# Define Hypermodel 

def build_model(hp):
    inputs = keras.Input(shape=input_shape + (1,))
    x = layers.Conv1D(
        filters=hp.Int('filters_1', min_value=32, max_value=128, step=32),
        kernel_size=hp.Choice('kernel_size_1', values=[3, 5]),
        activation="relu"
    )(inputs)
    x = layers.BatchNormalization()(x)
    x = layers.MaxPooling1D(2)(x)
    x = layers.Conv1D(
        filters=hp.Int('filters_2', min_value=64, max_value=256, step=64),
        kernel_size=hp.Choice('kernel_size_2', values=[3, 5]),
        activation="relu"
    )(x)
    x = layers.BatchNormalization()(x)
    x = layers.MaxPooling1D(2)(x)
    x = layers.Conv1D(
        filters=hp.Int('filters_3', min_value=128, max_value=512, step=128),
        kernel_size=hp.Choice('kernel_size_3', values=[3, 5]),
        activation="relu"
    )(x)
    x = layers.BatchNormalization()(x)
    x = layers.GlobalAveragePooling1D()(x)
    x = layers.Dense(
        units=hp.Int('dense_units', min_value=128, max_value=512, step=128),
        activation="relu"
    )(x)
    outputs = layers.Dense(4, activation="softmax")(x)
    model = keras.Model(inputs=inputs, outputs=outputs)
    
    model.compile(
        optimizer=keras.optimizers.Adam(
            hp.Float('learning_rate', min_value=1e-4, max_value=1e-2, sampling='LOG')
        ),
        loss=keras.losses.CategoricalCrossentropy(),
        metrics=["accuracy"]
    )
    return model

# Initialize the results list
res = []

# Iterate over each class
for l in [0, 1, 2, 3]:
    # Convert true labels and predicted labels to binary for the current class
    y_true_binary = (np.array(y_true_labels) == l).astype(int)
    y_pred_binary = (np.array(y_pred_labels) == l).astype(int)

    # Calculate precision, recall, fscore, and support
    precision, recall, _, support = precision_recall_fscore_support(
        y_true_binary, y_pred_binary, pos_label=1, average=None
    )

    # Sensitivity (recall for the positive class) is recall[1]
    sensitivity = recall[1]
    # Specificity (recall for the negative class) is recall[0]
    specificity = recall[0]

    # Append the results
    res.append([l, specificity, sensitivity])

# Create a DataFrame from the results

import matplotlib.pyplot as plt

def plot_confusion_matrix(cm,  classes, title, normalize=False, cmap=plt.cm.Blues):
    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title, fontname='Arial', fontsize=14, fontweight='bold')
    plt.colorbar()
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes, rotation=45, fontstyle='italic', fontname='Arial', fontsize=14)
    plt.yticks(tick_marks, classes, fontstyle='italic', fontname='Arial', fontsize=14)

    if normalize:
        cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]*100
        print("Normalized confusion matrix")
    else:
        print('Confusion matrix, without normalization')

    thresh = cm.max() / 2.
    for i, j in itertools.product(range(cm.shape[0]), range(cm.shape[1])):
        if normalize:
            plt.text(j, i, f"{cm[i, j]:.1f}%",
                     horizontalalignment="center",
                     color="white" if cm[i, j] > thresh else "black")
        else:
            plt.text(j, i, f"{int(cm[i, j])}",
                     horizontalalignment="center",
                     color="white" if cm[i, j] > thresh else "black")
    plt.tight_layout()
    plt.ylabel('True label', fontname='Arial', fontsize=12)
    plt.xlabel('Predicted label', fontname='Arial', fontsize=12, labelpad=-1)
    
    plt.savefig('confusion_matrix.png', dpi=600)

def plot__comb_confusion_matrix(cm, classes, title, cmap=plt.cm.Blues):
    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title, fontname='Arial', fontsize=14, fontweight='bold')
    plt.colorbar()
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes, rotation=45, fontstyle='italic', fontname='Arial', fontsize=14)
    plt.yticks(tick_marks, classes, fontstyle='italic', fontname='Arial', fontsize=14)

    cm_normalized = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis] * 100
    print("Normalized confusion matrix")

    thresh = cm.max() / 2.
    for i, j in itertools.product(range(cm.shape[0]), range(cm.shape[1])):
        plt.text(j, i, f"{int(cm[i, j])}\n({cm_normalized[i, j]:.1f}%)",
                 horizontalalignment="center",
                 verticalalignment="center",
                 color="white" if cm[i, j] > thresh else "black",
                 fontsize=10)

    plt.tight_layout()
    plt.ylabel('True label', fontname='Arial', fontsize=12)
    plt.xlabel('Predicted label', fontname='Arial', fontsize=12, labelpad=-1)

    plt.savefig('confusion_matrix_normcombined.png', dpi=600)

    y_pred = estimator.predict(X)
    return accuracy_score(np.argmax(y_true, axis=1), np.argmax(y_pred, axis=1))
